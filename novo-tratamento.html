function salvarTratamento() { var clienteNome = document.getElementById('clienteInput').value.trim(); var clienteId = document.getElementById('clienteId').value.trim(); if (!clienteNome) { mostrarMsg('Selecione um cliente.', 'error'); return; } if (!clienteId) { for (var i = 0; i < listaClientes.length; i++) { var x = listaClientes[i]; if ((getVal(x, 'nome', 'name', 'nomecliente', 'cliente') || '').toLowerCase() === clienteNome.toLowerCase()) { clienteId = getVal(x, 'id', 'idcliente', 'clienteid', 'codigo', 'cod') || ''; break; } } } if (!clienteId) { mostrarMsg('Cliente nao encontrado.', 'error'); return; } if (!graficos.length) { mostrarMsg('Adicione ao menos um grafico.', 'error'); return; } var tratamentoId = tratamentoSelecionadoId || document.getElementById('tratamentoId').value; var payload = { tipo: 'tratamento', action: 'criar_tratamento', cliente_id: clienteId, cliente_nome: clienteNome, tratamento_id: tratamentoId, atendimento_id: document.getElementById('idAtendimentoSelect').value || '', data_inicio: document.getElementById('dataInicio').value, graficos: [] }; for (var i = 0; i < graficos.length; i++) { var g = graficos[i]; payload.graficos.push({ ordem: i + 1, grafico: g.grafico, data_inicio: g.data_inicio, data_fim: g.data_fim, status: g.status }); } mostrarMsg('Salvando...', 'success'); fetch(URL_ATEND, { method: 'POST', body: JSON.stringify(payload), redirect: 'follow' }) .then(function(r) { return r.text(); }) .then(function(text) { var webhooks = []; for (var i = 0; i < graficos.length; i++) { var g = graficos[i]; if (g.data_fim && g.grafico) { var wh = { action: 'criar', calendario: 'graficos', cliente: clienteNome, cliente_id: clienteId, grafico: g.grafico, data_inicio: g.data_inicio, data_fim: g.data_fim, status: g.status, tratamento_id: payload.tratamento_id }; webhooks.push(fetch(URL_WEBHOOK, { method: 'POST', body: JSON.stringify(wh), redirect: 'follow' }).then(function(r2) { return r2.text(); }).catch(function() {})); } } Promise.all(webhooks).then(function() { mostrarMsg('Tratamento salvo! ID: ' + payload.tratamento_id, 'success'); setTimeout(function() { window.location.href = 'lista-tratamentos.html'; }, 1500); }); }) .catch(function(err) { mostrarMsg('Erro: ' + err.message, 'error'); }); }
